\begin{frame}{明文存金额会出什么事}
\begin{itemize}
  \item \textbf{合规}：等保 2.0 / PCI-DSS 要求"敏感字段不得明文落盘"，金额属于此类。
  \item \textbf{防内鬼}：DBA / 离职工程师拿到一次 dump → 全站余额裸了。
  \item \textbf{防误日志}：明文金额一旦被 \texttt{log.Debug(user)} 序列化 → 上传到 ELK。
  \item \textbf{审计}：金额被 SQL 直改 → 密文校验失败可立刻发现。
\end{itemize}
\keypoint{加密金额不是"加密功能"，是\textbf{把金额从数据库视野里抹掉}。运维看到的永远是乱码。}
\end{frame}

// EncryptMoney 加密金额
func (u *User) EncryptMoney(key string) (money string, err error) {
    aesObj, err := secret.NewAesEncrypt(
        conf.Config.EncryptSecret.MoneySecret, // 服务端持有
        key,                                   // 用户支付密码
        "", secret.AesEncrypt128, secret.AesModeTypeCBC)
    if err != nil { return }
    money = aesObj.SecretEncrypt(u.Money)
    return
}

// DecryptMoney 返回值单位为分
func (u *User) DecryptMoney(key string) (money int64, err error) {
    aesObj, _ := secret.NewAesEncrypt(
        conf.Config.EncryptSecret.MoneySecret, key, "",
        secret.AesEncrypt128, secret.AesModeTypeCBC)
    money = cast.ToInt64(aesObj.SecretDecrypt(u.Money))
    return
}
\end{lstlisting}
\end{frame}

\begin{frame}{逐行讲解：加解密做了什么}
\begin{itemize}\small
  \item \textbf{61 行}：服务端密钥 + 用户密码当两个独立输入，任一方旋转都让原密文失效。
  \item \textbf{64 行 CBC 而非 ECB}：同一笔金额每次密文不同，防"对照表"猜金额。
  \item \textbf{65 行 \texttt{Money string}}：密文是 base64 长度可变，直接存 string。
  \item \textbf{77 行 \texttt{cast.ToInt64}}：解密回 int64，\textbf{单位是分} —— 全链路避免浮点。
  \item \textbf{错误传播}：密码错 → 乱码 → \texttt{ToInt64} 返 0 → 金额校验失败 → TX 回滚。
\end{itemize}
\srcnote{repository/db/model/user.go:59--79}
\end{frame}

\begin{frame}{支付密码 vs 登录密码：为什么分两套}
\small
\begin{tabular}{@{}lll@{}}
\toprule
        & 登录密码     & 支付密码     \\
\midrule
作用    & 拿 JWT      & 解密金额     \\
长度    & $\geq 6$    & 固定 6 位    \\
泄漏    & 看订单地址  & 直接扣款     \\
隔离    & bcrypt 单向 & AES 双向     \\
存储    & 哈希        & 不存(仅入参) \\
\bottomrule
\end{tabular}

\begin{itemize}
  \item 登录密码哈希存：服务端只校验、不还原。
  \item 支付密码\textbf{不存}：用户每次手输 → 整库 dump 也解不开。
  \item 入口校验 \texttt{len(req.Key) == 6}，太短直接拒。
\end{itemize}

\begin{frame}{逐行讲解：扣款这一步的三个要点}
\begin{itemize}
  \item \textbf{57 行 UnPaid 校验在 TX 内}：不能 TX 外读再进事务 —— 否则典型 TOCTOU，两并发支付都看到 UnPaid → 双扣。
  \item \textbf{65 行 \texttt{DecryptMoney(req.Key)}}：密码错 → 解出乱码 → \texttt{userMoney} 几乎一定不够 → 走"金额不足"分支。\textbf{密码错和余额不足表象一致}，防穷举。
  \item \textbf{71--72 行 重新加密}：\texttt{Money} 是 string 不能直接减；必须 decrypt $\to$ int64 减 $\to$ encrypt。整段在 TX 里，失败回滚干净。
  \item \textbf{金额不足也 \texttt{return err}}：让 TX 走 rollback，\texttt{order.Type} 不会推进，客户端可原 OrderId 重试。
\end{itemize}
\srcnote{service/payment.go:55--96}

\section{第三方支付与熔断}
% ============================================================

\begin{frame}{当前 gomall 的"第三方支付"是什么}
\begin{itemize}
  \item gomall 当前未接真实支付宝/微信，"扣买家、入卖家"在本地 DB 完成。
  \item 但 \texttt{/paydown} 已挂 \texttt{CircuitBreaker} —— \textbf{给未来留位置}：
  \begin{itemize}
    \item 接入真实网关后链路变成"DB tx + 同步 RPC 第三方"。
    \item 第三方是慢依赖：p99 数百 ms + 会"集体抽风"。
    \item 不熔断 → goroutine 堆积 → 进程 OOM → 全站崩。
  \end{itemize}
  \item 熔断作用：\textbf{宁可 5\% 用户立即看到"稍后重试"，也不让 100\% 用户白屏 30s}。
\end{itemize}
\srcnote{routes/router.go:114--121；service/payment.go 当前实现}
\end{frame}

\begin{frame}[fragile]{allow() —— 决定能否放行}
\begin{lstlisting}[language=Go,firstnumber=76,
  caption={\srcnote{middleware/circuitbreaker.go:76--106}}]
func (cb *circuitBreaker) allow() error {
    switch circuitState(cb.state.Load()) {
    case stateClosed:
        return nil                            // 热路径零开销
    case stateOpen:
        if time.Now().UnixNano()-cb.openedAt.Load() < int64(cb.opt.OpenTimeout) {
            return errCircuitOpen             // 未到自愈窗口
        }
        cb.mu.Lock(); defer cb.mu.Unlock()
        if circuitState(cb.state.Load()) == stateOpen { // double-check
            cb.state.Store(int32(stateHalfOpen))         // Open -> HalfOpen
            cb.halfOpenReq.Store(0)
        }
        if cb.halfOpenReq.Add(1) > cb.opt.HalfOpenMaxReq {
            return errCircuitOpen             // 探测配额满
        }
        return nil
    }
    return nil
}
\end{lstlisting}
\end{frame}

\begin{frame}{逐行讲解：allow() 的关键决策}
\begin{itemize}
  \item \textbf{Closed 分支零开销}：只一次原子 Load → return nil，不抢锁不计数。这是熔断器能放 paydown 入口的前提。
  \item \textbf{Open 早返回}：\texttt{time.Now() - openedAt} 无锁原子读，99\% 的 Open 期请求从这里返回。
  \item \textbf{加锁切 HalfOpen}：只有自愈窗口的\textbf{第一个}请求加锁切状态 —— double-check 防两 goroutine 同时切。
  \item \textbf{HalfOpen 探测配额}：\texttt{halfOpenReq.Add(1) > 3} 保证半开期最多放 3 个试水。
  \item \textbf{report() 配套}：3 个全成功 → 回 Closed；任一失败 → 立即重新 Open，时钟重置。
\end{itemize}
\srcnote{middleware/circuitbreaker.go:76--128}
\end{frame}

\begin{frame}[fragile]{失败的定义 + 阈值 5 / 10s / 3}
\begin{lstlisting}[language=Go,firstnumber=69,caption={\srcnote{middleware/circuitbreaker.go:69--70}}]

\begin{frame}[fragile]{Idempotency 中间件的 responseRecorder}
\begin{lstlisting}[language=Go,firstnumber=17,caption={\srcnote{middleware/idempotency.go:17--31, 96--110}}]
// responseRecorder 拷贝写入的响应体，便于幂等命中时回放
type responseRecorder struct {
    gin.ResponseWriter
    body *bytes.Buffer
}
func (r *responseRecorder) Write(b []byte) (int, error) {
    r.body.Write(b)                        // 旁路缓存
    return r.ResponseWriter.Write(b)       // 同时真写出去
}

// 主流程末尾:
recorder := &responseRecorder{ResponseWriter: c.Writer, body: bytes.NewBuffer(nil)}
c.Writer = recorder
c.Next()
if len(c.Errors) > 0 || recorder.Status() >= http.StatusBadRequest {
    cache.ReleaseIdempotencyLock(c.Request.Context(), key)   // 失败: 回 init
    return
}
cache.CommitIdempotencyResult(c.Request.Context(), key, recorder.body.String())
\end{lstlisting}
\end{frame}

\begin{frame}{逐行讲解：responseRecorder 解决了什么}
\begin{itemize}
  \item \textbf{嵌入 gin.ResponseWriter}：包一层而非替换，\texttt{Header/Status/Hijack} 透明转发，链路下游无感。
  \item \textbf{双写}：先写 \texttt{bytes.Buffer} 再调原 \texttt{Writer.Write} —— 顺序很重要，反过来客户端断连缓存会写不完。
  \item \textbf{Hook 时机}：在 \texttt{c.Next()} 之前替换 Writer，业务函数写入时被拦截。
  \item \textbf{失败回滚}：4xx 或 \texttt{c.Errors} → 回退到 init → 同 IK 还能再试（业务失败 ≠ 永久失败）。
  \item \textbf{成功提交}：把完整 JSON 写进 Redis done 态 → 后续同 IK 由 Lua 直接回放，业务代码不再执行。
\end{itemize}
\srcnote{middleware/idempotency.go:17--112}

\begin{frame}{Idempotency 的极限场景：755K 请求 → 1 笔订单}
\begin{itemize}
  \item 50 VU $\times$ 15s 用\textbf{同一个 IK} 反复打 \texttt{/orders/create}。
  \item 累计 \textbf{755{,}033 次请求}，DB 实际订单数 \textbf{= 1}：
  \begin{itemize}
    \item \texttt{SELECT COUNT(*) FROM `order` WHERE user\_id=10 AND created\_at > NOW() - 2 MIN} → 1
  \end{itemize}
  \item 第一次走完业务后，后续全部由 Lua 读 \texttt{done} 态直接回放 → 50K RPS。
  \item \textbf{业务承诺}：客户端在熔断反复触发期间无脑重试 10K 次，仍只扣一次款。
\end{itemize}
\keypoint{重试是客户端的权利，幂等是服务端的义务。两者共同维护"恰好一次"。}
\srcnote{stressTest/REPORT.md \S 4 幂等中间件压测}
\end{frame}

\begin{frame}{面试 Q\&A}
\small
\textbf{Q1}：金额为什么不直接 \texttt{decimal} 存？\\
\hspace{4mm} A：合规要求"敏感字段不得明文落盘"。AES + 服务端密钥 + 用户密码双因子，比 decimal 多一道防线。

\vspace{1mm}
\textbf{Q2}：熔断打开期间客户端怎么处理？\\
\hspace{4mm} A：拿 70002 → 指数退避 (1s/2s/4s) → 重试\textbf{必须带原 Idempotency-Key}。第一次若其实成功，第二次回放原结果不会二次扣款。

\vspace{1mm}
\textbf{Q3}：为何不用 Redis 分布式锁做幂等，要用 Lua？\\
\hspace{4mm} A：分布式锁只解决"不重入"，不解决"结果回放"。Lua 是真状态机 (init/processing/done)：第一次走完后，同 IK 请求直接 Lua 内回放，\textbf{不进业务代码}。

\vspace{1mm}
\textbf{Q4}：密码错和余额不足为什么都返回"金额不足"？\\
\hspace{4mm} A：密码错 → 乱码 → int64 几乎不够 → 走金额不足分支。\textbf{有意为之}：防穷举区分。
\end{frame}
