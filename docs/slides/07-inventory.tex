\section{为什么单桶 stock 不够用}
% ============================================================

\begin{frame}{下单到付款，中间那 15 分钟}
\begin{itemize}
  \item 用户点"提交订单"到点"立即支付"，gomall 给 15 分钟。
  \item 这 15 分钟里：商品\textbf{不能再被别人下单}，但也\textbf{还没真正卖出}。
  \item 单桶 \texttt{stock=N} 模型无法表达这个"占着但没消耗"的中间态：
  \begin{itemize}
    \item 一减就减，用户不付钱也回不来（库存丢）；
    \item 不减，并发下两个人同时下同一件（超卖）。
  \end{itemize}
\end{itemize}
\srcnote{repository/cache/inventory.go:11--14 注释明确两桶定位}
\end{frame}

\begin{frame}{业务问题清单（本 deck 七连问）}
\begin{enumerate}
  \item 为什么把库存切成 \textbf{available + reserved} 两桶？
  \item 抢库存为什么必须走 Redis Lua，不走 DB \texttt{SELECT FOR UPDATE}？
  \item Redis 重启之后，库存数字从哪儿来？
  \item 下单成功，但 DB 事务回滚，预占的库存怎么办？
  \item 关单（超时 / 用户主动）触发的库存退回，落在哪一行代码？
  \item 客服收到"明明显示有货，下单却失败"的工单，怎么回答？
  \item 业务到底承诺什么 —— 超卖能赔多少钱、误锁赔多少钱？
\end{enumerate}

\section{Redis Lua 三段脚本}
% ============================================================

\begin{frame}{三态生命周期}
\begin{tikzpicture}[node distance=10mm and 18mm]
  \node[node-state,node-state-normal] (av)  {available};
  \node[node-state,node-state-transition,right=of av] (rv) {reserved};
  \node[node-state,node-state-alert,right=of rv] (out) {sold};
  \draw[arrow] (av) -- node[above,font=\tiny,align=center]{reserve\\(下单)} (rv);
  \draw[arrow] (rv) -- node[above,font=\tiny,align=center]{commit\\(支付)} (out);
  \draw[arrow,bend left=35,draw=cOK] (rv.north) to node[above,font=\tiny\color{cOK},align=center]{release\\(取消/超时)} (av.north);
\end{tikzpicture}
\vspace{2mm}
\begin{itemize}
  \item \textbf{reserve}：available -= n, reserved += n（下单时调）
  \item \textbf{commit}：reserved -= n（支付成功时调，total 也消耗）
  \item \textbf{release}：reserved -= n, available += n（取消 / 关单时调）
\end{itemize}
\srcnote{repository/cache/inventory.go:29--66 三段 Lua 脚本完整定义}
\end{frame}

\begin{frame}[fragile]{reserveScript：8 行 Lua，零超卖的全部秘密}
\begin{lstlisting}[language=LuaRedis,firstnumber=32,
  caption={\srcnote{repository/cache/inventory.go:32--44 reserveScript}}]
local avail = redis.call('GET', KEYS[1])    -- 1. 读 available
if avail == false then
    return -2                                -- 2. key 不存在 = 库存未初始化
end
local need = tonumber(ARGV[1])
if tonumber(avail) < need then
    return -1                                -- 3. 不足 = ErrStockInsufficient
end
redis.call('DECRBY', KEYS[1], need)          -- 4. available -= n
redis.call('INCRBY', KEYS[2], need)          -- 5. reserved  += n
return 1                                     -- 6. 成功
\end{lstlisting}
\small \textbf{为什么必须打成一个 Lua}：脚本对 Redis 是单命令，redis-server 单线程从头执行到尾，中途\textbf{不会插入}其它客户端的 GET / DECRBY / INCRBY。如果分成 3 次 RTT 做 check-then-act，500 并发立刻超卖。
\end{frame}

\begin{frame}[fragile]{commitScript / releaseScript：成对出现，对称设计}
\begin{lstlisting}[language=LuaRedis,firstnumber=48,
  caption={\srcnote{repository/cache/inventory.go:48--66 commit + release}}]
-- commitScript：支付成功，reserved 真正消耗
local r = redis.call('GET', KEYS[1])           -- KEYS[1]=reserved
if r == false or tonumber(r) < tonumber(ARGV[1]) then
    return -1
end
redis.call('DECRBY', KEYS[1], ARGV[1])         -- reserved -= n
return 1
-- releaseScript：取消，reserved 退回 available
local r = redis.call('GET', KEYS[1])           -- KEYS[1]=reserved
if r == false or tonumber(r) < tonumber(ARGV[1]) then
    return -1
end
redis.call('DECRBY', KEYS[1], ARGV[1])         -- reserved -= n
redis.call('INCRBY', KEYS[2], ARGV[1])         -- available += n
return 1
\end{lstlisting}
\small commit 只动 reserved 一桶（钱已付，不退）；release 两桶都动（占的位还回去）。每次调用前必 check $r \geq n$，避免 reserved 跌成负数。
\end{frame}

\end{itemize}

\begin{itemize}
  \item \textbf{真理之源}是 MySQL 的 \texttt{product.num}（商家后台维护，AOF / binlog 持久化）。
  \item Redis 只是\textbf{运行时缓存}，重启会丢，必须能从 DB 恢复。
  \item 但已经在跑的实例\textbf{不能}被覆盖，否则会把 reserved 状态吃掉。
\end{itemize}

\end{frame}

\begin{frame}[fragile]{SeedFromDB 主循环：分页 + EXISTS 双保险}
\begin{lstlisting}[language=Go,firstnumber=14,
  caption={\srcnote{service/inventory/syncer.go:14--48 SeedFromDB}}]
func SeedFromDB(ctx context.Context, batchSize int) error {
    if batchSize <= 0 { batchSize = 500 }            // 默认页 500
    d := dao.NewDBClient(ctx); var lastID uint
    for {
        var rows []*model.Product
        q := d.Model(&model.Product{}).Order("id ASC").Limit(batchSize)
        if lastID > 0 { q = q.Where("id > ?", lastID) }  // 游标分页
        if err := q.Find(&rows).Error; err != nil { return err }
        if len(rows) == 0 { break }
        for _, p := range rows {
            lastID = p.ID
            exists, _ := cache.RedisClient.Exists(ctx,
                cache.StockAvailableKey(p.ID)).Result()
            if exists == 1 { continue }              // 跳过运行期数据
            cache.InitStock(ctx, p.ID, int64(p.Num)) // 覆盖写 0
        }
    }
    return nil
}
\end{lstlisting}
\end{frame}

\begin{frame}{为什么是"游标分页 + EXISTS 跳过"}
\begin{itemize}
  \item \textbf{游标分页}（\texttt{WHERE id > lastID LIMIT 500}）：
  \begin{itemize}
    \item 1M 行的 \texttt{product} 表，\texttt{OFFSET 999500 LIMIT 500} 要扫近百万行；
    \item 游标走 PK 索引，每页\textbf{常数时间}，启动期间 DB CPU 不抖。
  \end{itemize}
  \item \textbf{EXISTS 检查 / 跳过}（line 32--40 是关键）：
  \begin{itemize}
    \item 冷启动：available key 不存在 $\to$ 写入 DB 水位；
    \item 滚动重启 / 多副本：available 已被前一实例 / 上一次启动写过 $\to$ 跳过，\textbf{不会把运行期 reserve 抹平}；
    \item \texttt{InitStock} 是\textbf{覆盖写}（\texttt{SET k v 0}），没这层 EXISTS 就是事故。
  \end{itemize}
\end{itemize}
\srcnote{service/inventory/syncer.go:34--41 EXISTS 跳过}
\end{frame}

\begin{frame}[fragile]{CancelUnpaidOrder 的幂等保证}
\begin{lstlisting}[language=Go,firstnumber=31,
  caption={\srcnote{service/order\_cancel.go:31--62 关单 + release}}]
err = baseDao.DB.Transaction(func(tx *gorm.DB) error {
    ok, err := dao.NewOrderDaoByDB(tx).CloseOrderWithCheck(orderNum)
    if err != nil { return err }

\section{业务承诺与客诉答疑}

\begin{frame}{业务承诺：超卖 vs 误锁，两条不同的红线}
\begin{tabular}{@{}lll@{}}
\toprule
事故 & 用户体感 & 业务代价 \\
\midrule
超卖     & 下单成功，无法发货      & \textbf{真金白银赔} + 商誉 \\
误锁     & 显示"已售罄"           & 转化率轻微下降 \\
\bottomrule
\end{tabular}
\vspace{3mm}
\begin{itemize}
  \item 超卖 = 法律 / 合同事件，必须\textbf{零容忍}（NoOversell 单测就是这一条的工程化）。
  \item 误锁 = 体验事件，可以\textbf{容忍一阵}（高峰过后 / 关单超时后会自愈）。
  \item 这个不对称是\textbf{所有设计选择}的源头：选 Redis 单线程、选先 reserve 后 commit、选 EXISTS 跳过 \dots
\end{itemize}
\srcnote{repository/cache/inventory.go:11--14 注释揭示这条业务红线}
\end{frame}

\begin{frame}{客服话术：50001 vs 70001 vs 70002}
\begin{tabular}{@{}lll@{}}
\toprule
状态码 & 触发位置 & 客服话术 \\
\midrule
\textbf{50001} & ReserveStock 返回 ErrStockInsufficient & "本商品已售罄" \\
70001 & 全局 / 秒杀限流命中 & "活动太火爆，请稍后" \\
70002 & 支付熔断器 Open & "支付服务暂忙，1 分钟后" \\
60002 & 同 key 重复下单 & "您已下单成功" \\
\bottomrule
\end{tabular}
\vspace{2mm}
\small 50001 和 70001 都是"拒单"，但根因不同：
\begin{itemize}
  \item 50001 是\textbf{真的}没货（available = 0），重试也没用；
  \item 70001 是\textbf{速率}被限，等几秒重试就行。
\end{itemize}
\srcnote{pkg/e/code.go 业务码定义；50001 在 deck 10 业务码总表中作为库存不足语义}
\end{frame}

\begin{frame}{典型客诉：明明显示有货，下单却失败}
\textbf{Q：商详页显示库存 50，我下单却提示"已售罄"？}

\vspace{2mm}
两种根因：
\begin{itemize}
  \item \textbf{商详页缓存 vs Redis available 不一致}：
  \begin{itemize}
    \item 商详页用 5min 商品缓存（deck 03），num 字段是 5 分钟前的快照；
    \item 高峰期，available 已经从 50 跌到 0，但商详页还显示 50；
    \item 应对：商详页改成显示"是否有货" boolean，或缓存 TTL 调到 30s。
  \end{itemize}
  \item \textbf{库存被预占未释放}：
  \begin{itemize}
    \item 大量未支付订单堆在 reserved 桶，available = 0；
    \item 15 min 后被关单 worker 释放回去；
    \item 应对：关单 worker 心跳监控，延迟超过 2min 报警。
  \end{itemize}
\end{itemize}
\srcnote{service/order\_cancel.go:59 release 调用是这条客诉自愈的关键}
\end{frame}

\begin{frame}{巡检对账 + 各角色看哪个指标}
\textbf{巡检恒等式}：\texttt{available + reserved + committed = product.num}\\
\small \texttt{GetStockSnapshot(pid)} 给前两项，committed 取 DB 已支付订单 \texttt{SUM(num)}；偏差 $>$ 1\% 钉钉报警，SRE 用 \texttt{InitStock} 人工覆盖，\textbf{不阻断业务}。
\vspace{1mm}
\begin{tabular}{@{}lll@{}}
\toprule
角色 & 关心 & 看哪个 \\
\midrule
财务  & 不超卖     & NoOversell 单测 + 已发货 vs DB 水位差 \\

\begin{frame}{面试 Q\&A}
\textbf{Q1}：为什么不能用单条 \texttt{UPDATE product SET num=num-1 WHERE num>0}？\\
\hspace{4mm}\small A：单 SQL 防得了超卖，但表达不了"占着未付"。下单后到支付前的 15min，单桶模型只能选"立刻扣"（不付就丢货）或"不扣"（并发超卖），都不可接受。

\vspace{2mm}
\textbf{Q2}：Redis 挂了 30 秒，恢复后库存怎么办？\\
\hspace{4mm}\small A：AOF / RDB 持久化保住大头；丢失的部分由 SeedFromDB 在重启时从 \texttt{product.num} 回灌。运行期数据（reserved 状态）通过 EXISTS 跳过保护。少量已支付未对账的差额走巡检脚本对齐。

\vspace{2mm}
\textbf{Q3}：DB 事务成功但 Redis commit 失败，怎么处理？\\
\hspace{4mm}\small A：当前实现是 outbox 事件落库后由消费者重试 CommitReservation；reserved 桶里数字偏高，巡检 leak 报警接管。属"业务可接受 + 监控可发现"的设计取舍。
\end{frame}
