\vspace{2mm}
\begin{itemize}
  \item 进站用户里约 60\% 会主动用搜索，是漏斗最宽的一跳。
  \item "搜不到 / 搜偏" 用户不会换词，直接退出 App。
  \item 入口处的相关性，决定后面三跳还能不能继续。

\begin{frame}{业务对搜索的四条硬约定}
\begin{itemize}
  \item \textbf{召回}：用户搜得到的词，能跑出至少一条相关结果。
  \item \textbf{时延}：p95 < 200ms，列表页才不会"卡半秒空白"。
  \item \textbf{新鲜度}：商家上架 $\to$ 站内可搜，SLA $\le$ 5min。
  \item \textbf{兜底}：ES 挂掉，搜索接口不能 5xx，必须有降级路径。
\end{itemize}
\vspace{2mm}
\keypoint{这四条决定了搜索系统不能只是 \texttt{SELECT * WHERE name LIKE '\%kw\%'}。}
\srcnote{stressTest/REPORT.md：/product/show 同库 PK 查询基线 62{,}226 RPS / p95 3ms 作为时延锚点}
\end{frame}

% ============================================================

\begin{frame}{ES 替 LIKE 做什么}
\begin{tikzpicture}[node distance=6mm and 12mm]
  \node[node-box]                            (q)   {用户输入\\\textit{苹果手机}};
  \node[node-box,right=of q,node-warn]       (an)  {分词\\\textit{苹果} / \textit{手机}};
  \node[node-box,right=of an]                (idx) {倒排索引\\term $\to$ docs};
  \node[node-box,right=of idx,node-ok]       (sc)  {打分\\BM25 + boost};
  \node[node-box,below=of sc]                (out) {结果列表\\相关性降序};
  \draw[arrow] (q)   -- (an);
  \draw[arrow] (an)  -- (idx);
  \draw[arrow] (idx) -- (sc);
  \draw[arrow] (sc)  -- (out);
\end{tikzpicture}
\vspace{2mm}
\begin{itemize}
  \item \textbf{分词}：按词切（中文走 IK 或 standard fallback）。
  \item \textbf{倒排索引}：term 直接索引到 doc 列表，避开全表扫。
  \item \textbf{打分}：\texttt{multi\_match} + 字段加权，name \textgreater{} title \textgreater{} info。
\end{itemize}
\srcnote{repository/es/product\_index.go:163--227 SearchProductsWithScore 多字段加权查询}
\end{frame}

\begin{frame}{字段加权背后的业务判断}
\begin{tabular}{@{}llr@{}}
\toprule
字段          & 业务含义                       & 加权     \\
\midrule
\texttt{name}  & 商品主名（用户预期匹配）        & $\times 3$ \\
\texttt{title} & 商品标题（含型号 / 规格）       & $\times 2$ \\
\texttt{info}  & 商品详情（长文，噪声多）        & $\times 1$ \\
\bottomrule
\end{tabular}
\vspace{2mm}
\begin{itemize}
  \item 用户搜 "iPhone"，"标题就叫 iPhone 13" 应当排在"info 提到 iPhone"前面。
  \item 加权由运营调，不是开发拍。大促可临时把 \texttt{title} 上调到 $\times 4$。
  \item 不加权 $\to$ "标签店家把热词塞进 info" 会霸占第一屏。
\end{itemize}
\srcnote{repository/es/product\_index.go:174--177 \texttt{fields=\{name\^{}3,title\^{}2,info\}}}
\end{frame}

\begin{frame}[fragile]{代码段 1：ES 关键词检索（含 \_score 输出）}
\begin{lstlisting}[language=Go,firstnumber=170,basicstyle=\ttfamily\tiny]

    "multi_match": map[string]any{

    filter = append(filter, map[string]any{

    })

q := map[string]any{
    "from": from, "size": size,
    "query": map[string]any{"bool": map[string]any{

\end{lstlisting}
\srcnote{repository/es/product\_index.go:170--191 关键片段}
\end{frame}

\begin{frame}{代码段 1 逐行讲解}
\begin{itemize}
  \item \textbf{L170--175}：\texttt{multi\_match} 同时查 name / title / info，靠 \texttt{\^{}N} 给字段加权。这一行就是上一页那张加权表的代码落点。
  \item \textbf{L176--180}：\texttt{filter} 走 \texttt{term}，类目过滤不参与打分，只做 yes/no 过滤，性能更稳。运营按类目限定时进这条分支。
  \item \textbf{L182--189}：\texttt{bool.must + filter} 是 ES 的标准组合 —— must 决定相关性，filter 决定可见性。两者分离，分页和打分才不会互相干扰。
  \item \textbf{业务对应}：name 权重最高，对应"用户最在意主名命中"；filter 不打分，对应"类目是硬过滤，不是建议"。
\end{itemize}
\end{frame}

\section{上架到可搜：outbox + 异步索引}
% ============================================================

\begin{frame}{业务承诺：商家上架 5 分钟内必须搜得到}
\begin{itemize}
  \item 商家后台点"上架" $\to$ 站内搜索能搜到，业务侧的 SLA 是 $\le$ 5min。
  \item 5min 是商家容忍上限：拍照 / 调价 / 改文案的反馈节奏。
  \item 写 DB 同步写 ES 行不行？\textbf{不行}：ES 抖动直接 fail 上架，商家退出。
  \item 必须\textbf{解耦}：DB 写完就返回成功，ES 异步追上。
\end{itemize}
\srcnote{service/product.go:155, 253, 289 三处商品写入后调 emitProductChanged}
\end{frame}

\begin{frame}{outbox $\to$ MQ $\to$ indexer 的链路}
\begin{tikzpicture}[node distance=8mm and 10mm]

  \node[node-box,right=of api]               (db)  {MySQL\\product + outbox};
  \node[node-box,right=of db,node-warn]      (pub) {Outbox\\Publisher};

  \draw[arrow-async] (mq)  -- (idx);
  \draw[arrow]       (idx) -- (es);
\end{tikzpicture}
\vspace{2mm}
\begin{itemize}
  \item 上架事务里只写 product + outbox，二者同库同事务，原子。
  \item Publisher 后台扫 outbox $\to$ 投递到 \texttt{product.changed} 路由。
  \item Indexer 消费后写 ES。Qos=32 控制并发，避免 ES 被写穿。
\end{itemize}
\srcnote{service/product.go:257--265 emitProductChanged；service/search/indexer.go:14--49 indexer 主循环}
\end{frame}

\begin{frame}[fragile]{代码段 2：indexer 主循环}
\begin{lstlisting}[language=Go,firstnumber=17,basicstyle=\ttfamily\tiny]
func StartProductIndexer(ctx context.Context) error {
    rabbitmq.BindDomainQueue(indexerQueue, "product.changed")
    ch, _ := rabbitmq.GlobalRabbitMQ.Channel()
    ch.Qos(32, 0, false)  // 在途上限 32 条
    msgs, _ := ch.Consume(indexerQueue, "", false, false, false, false, nil)
    go func() {
        for d := range msgs {
            var ev events.ProductChanged
            if json.Unmarshal(d.Body, &ev) != nil {
                d.Nack(false, false); continue  // 脏消息进死信
            }
            if handleProductChanged(ctx, ev) != nil {
                d.Nack(false, true);  continue  // 重投自愈
            }
            d.Ack(false)
        }
    }()
    return nil
}
\end{lstlisting}
\srcnote{service/search/indexer.go:17--49（折行省略错误处理）}
\end{frame}

\begin{frame}{代码段 2 逐行讲解}
\begin{itemize}
  \item \textbf{L18--19}：\texttt{BindDomainQueue} 把 \texttt{product.changed} 路由绑到 \texttt{search.product.indexer} 队列。上架 / 改价 / 下架共用一条路由。
  \item \textbf{L22}：\texttt{Qos(32, 0, false)} 让消费端最多在途 32 条未 ack。这是给 ES 写入设的"流量阀"，对应 deck 05 的"在网关之外，每个消费者也要节流"。
  \item \textbf{L30--34}：JSON 解析失败 $\to$ \texttt{Nack(false, false)}，不重投。脏消息进死信，不阻塞健康消息。
  \item \textbf{L35--38}：业务处理失败 $\to$ \texttt{Nack(false, true)} 重投，给 ES 短抖动一次自愈机会。
  \item \textbf{L39}：成功才 ack，保证 at-least-once。索引是幂等的（同 docID 覆盖），重复消费无害。
\end{itemize}
\srcnote{service/search/indexer.go:25 Qos=32；:36/:42 Nack 语义}
\end{frame}

\begin{frame}{对应 ES 写入：UpsertProduct 是幂等的}
\begin{itemize}
  \item \texttt{op=create/update} $\to$ \texttt{UpsertProduct(p)}；\texttt{op=delete} $\to$ \texttt{DeleteProduct(id)}。
  \item \texttt{DocumentID = product.ID}：同 ID 多次写入覆盖，天然幂等，对应消费侧的"重投无害"。
  \item \texttt{Refresh: "false"}：不强制刷盘，让 ES 自己按 1s 刷新窗口批量落盘，吞吐优先。
  \item 5min SLA 实际由 RMQ 堆积长度兜底：Qos=32 下，95\% 场景 < 30s 可搜。
\end{itemize}
\srcnote{service/search/indexer.go:51--60 handleProductChanged；repository/es/product\_index.go:105--131 UpsertProduct}
\end{frame}

\begin{frame}{代码段 3 逐行讲解}
\begin{itemize}

\begin{frame}{面试 Q\&A}
\textbf{Q1}：为什么不让上架 API 直接写 ES，而要 outbox？\\
\hspace{4mm}\small A：DB 与 ES 不同事务，同步写要么失败回滚（伤体验），要么不一致（伤召回）。outbox 把"DB 写成功" 和 "ES 写成功" 两件事拆开，分别保证。

\vspace{2mm}
\textbf{Q2}：ES 挂掉，搜索接口为什么不直接 5xx？\\
\hspace{4mm}\small A：搜索是入口，5xx 等于 app 崩页。LIKE 慢一点也比白屏强 —— 业务承诺"搜索框永远可用"。

\vspace{2mm}
\textbf{Q3}：hybrid 50/50 这个权重你怎么调？\\
\hspace{4mm}\small A：业务值，不是技术值。先 50/50 跑一轮，A/B 观察 CTR / 转化。短 query 偏关键词，长 query 偏语义。
\end{frame}
