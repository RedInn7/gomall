\section{第一层：TokenBucket 全局兜底}
% ============================================================

\begin{frame}{为什么全局还要 100 RPS / 200 burst}
\begin{itemize}
  \item gomall 单机基线 \textbf{64{,}254 RPS}（\texttt{/ping}），单 IP 真正用户用不到 1/600。
  \item 100 RPS / 200 burst 实际拦的是\textbf{爬虫脚本和异常客户端}，不是用户。
  \item burst 给到 2$\times$ rps 的依据：
  \begin{itemize}
    \item 浏览器一个页面 30+ 静态 GET 是常态，瞬时 burst $\approx$ 100；
    \item burst 留 200 = 让一次正常加载\textbf{绝不触发}。
  \end{itemize}
  \item 这层是\textbf{兜底}：业务层的精细限流挂掉时，至少不会被压死。
\end{itemize}
\srcnote{routes/router.go:22 TokenBucket(rate.Limit(100), 200)}
\end{frame}

\begin{frame}{TokenBucket 算法：桶 + 令牌生成 + 漏出}
\begin{tikzpicture}[node distance=8mm and 14mm]
  \node[node-box,node-warn] (gen) {令牌生成\\100/s 稳定};
  \node[node-box,right=of gen] (bucket) {桶\\容量 200};
  \node[node-box,right=of bucket,node-ok] (out) {请求漏出};
  \node[node-box,below=of bucket,node-bad] (drop) {满了丢弃\\status=70001};

  \draw[arrow] (bucket) -- (out);

\begin{itemize}
  \item 令牌按 \textbf{rps} 匀速生成；桶满则溢出。
  \item 请求拿到令牌才放行，拿不到立刻返回 70001。
  \item burst 容量决定能否吸收\textbf{瞬时尖刺}（比如页面首屏并发）。
\end{itemize}

\end{frame}

\begin{frame}[fragile]{TokenBucket 中间件实现 (逐行)}
\begin{lstlisting}[language=Go,firstnumber=22,basicstyle=\ttfamily\tiny,
  caption={\srcnote{middleware/ratelimit.go:22--49}}]
func TokenBucket(rps rate.Limit, burst int) gin.HandlerFunc {
    var mu sync.Mutex
    limiters := make(map[string]*rate.Limiter)
    get := func(ip string) *rate.Limiter {
        mu.Lock(); defer mu.Unlock()
        if l, ok := limiters[ip]; ok { return l }
        l := rate.NewLimiter(rps, burst); limiters[ip] = l; return l
    }
    return func(c *gin.Context) {
        if !get(c.ClientIP()).Allow() {
            c.JSON(200, gin.H{"status": e.ErrRateLimitExceeded})
            c.Abort(); return
        }
        c.Next()
    }
}
\end{lstlisting}
\scriptsize ① map 按 IP 隔离；② 每 IP 一个 \texttt{rate.Limiter}；③ \texttt{Allow()} 非阻塞；④ 拒了直接 70001；⑤ 进程内 map 是单机代价，多实例要 Redis token bucket。
\end{frame}

\section{第二层：SlidingWindow 用户精确限流}
% ============================================================

\begin{frame}{秒杀为什么必须用 SlidingWindow}
\begin{itemize}
  \item 秒杀的\textbf{业务对手}是脚本，不是用户：
  \begin{itemize}
    \item 真人最快点击 \textbf{300--500ms 一次} $\to$ 1s 最多 2--3 次；
    \item 脚本可以 1s 打 1000 次。
  \end{itemize}
  \item 阈值定 \textbf{3/s/用户}：
  \begin{itemize}
    \item 真用户：留 50\% 余量给"双击 + 网络重试"；
    \item 黑产：第 4 次开始 70001 截断。
  \end{itemize}
  \item TokenBucket 不够：
  \begin{itemize}
    \item 必须按\textbf{用户 ID}（不是 IP）限，否则一个 NAT 出口下所有用户共享一个桶；
    \item 必须\textbf{分布式精确}，多实例下用 Redis 共享状态。
  \end{itemize}
\end{itemize}
\srcnote{routes/router.go:136--142 SlidingWindow Scope:seckill Limit:3 Window:1s ByUser:true}
\end{frame}

\begin{frame}[fragile]{SlidingWindow Lua 脚本 (逐行)}
\begin{lstlisting}[language=LuaRedis,firstnumber=18,basicstyle=\ttfamily\tiny,
  caption={\srcnote{repository/cache/ratelimit.go:18--34}}]
local key, window_ms = KEYS[1], tonumber(ARGV[1])
local limit, now_ms, member = tonumber(ARGV[2]), tonumber(ARGV[3]), ARGV[4]
redis.call('ZREMRANGEBYSCORE', key, 0, now_ms - window_ms)
local count = redis.call('ZCARD', key)
if count >= limit then return {0, count} end
redis.call('ZADD', key, now_ms, member)
redis.call('PEXPIRE', key, window_ms)
return {1, count + 1}
\end{lstlisting}
\small 逐行：① 入参 key/窗口/阈值/now/请求 id；② \textbf{ZREMRANGEBYSCORE} 滑掉过期（关键）；③ ZCARD 数当前窗口；④ 超阈值返回 0；⑤ 通过则 ZADD 入；⑥ PEXPIRE 自动回收；⑦ \textbf{整脚本单原子}，无竞态。
\end{frame}

\section{第三层：CircuitBreaker 保护慢依赖}
% ============================================================

\begin{frame}{熔断保护的不是自己，是下游}
\begin{itemize}
  \item gomall \texttt{/paydown} 的下游 = 第三方支付（Stripe / 微信 / 支付宝）。
  \item 这种依赖的\textbf{失败模式}是慢，不是 500：
  \begin{itemize}
    \item 正常 RT 200ms；
    \item 异常时 RT 30s 才超时；
    \item 不熔断 $\to$ goroutine 堆积 $\to$ 内存爆 $\to$ 自己跟着挂。
  \end{itemize}
  \item CircuitBreaker 的业务含义：\textbf{"我代下游回答你不行，免得你等 30 秒"}。
  \item 三态阈值 \textbf{5 失败 / 10s 等待 / 3 探测}（router.go:114--119）：
  \begin{itemize}
    \item 5 = 容忍偶发抖动，不会一抖就跳；
    \item 10s = 给下游一个完整告警 + 自愈周期；
    \item 3 = 探测窗口够小，避免 HalfOpen 也被打爆。
  \end{itemize}
\end{itemize}
\srcnote{routes/router.go:114--119 三态阈值}
\end{frame}

\begin{frame}[fragile]{CircuitBreaker.allow 状态转移 (逐行)}
\begin{lstlisting}[language=Go,firstnumber=76,basicstyle=\ttfamily\tiny,
  caption={\srcnote{middleware/circuitbreaker.go:76--106 allow}}]
func (cb *circuitBreaker) allow() error {
    switch circuitState(cb.state.Load()) {
    case stateClosed: return nil                          // 1) 关闭态零开销
    case stateOpen:
        if time.Now().UnixNano()-cb.openedAt.Load() < int64(cb.opt.OpenTimeout) {
            return errCircuitOpen                         // 2) Open 期间一律拒
        }
        cb.mu.Lock(); defer cb.mu.Unlock()
        if circuitState(cb.state.Load()) == stateOpen {   // 3) 锁内 double-check
            cb.state.Store(int32(stateHalfOpen))
            cb.halfOpenReq.Store(0)
        }
        fallthrough                                       // 4) 进入 HalfOpen
    case stateHalfOpen:
        if cb.halfOpenReq.Add(1) > cb.opt.HalfOpenMaxReq {
            return errCircuitOpen                         // 5) 探测超限退回
        }
        return nil
    }
    return nil
}
\end{lstlisting}
\scriptsize ① Closed 仅 atomic load；② Open 内不查时间就拒；③ Open$\to$HalfOpen 锁内 double-check；④ HalfOpen 用原子 Add 限并发；⑤ 热路径无 mutex。
\end{frame}

\begin{frame}[fragile]{report：决定下一刻是 Closed 还是 Open}
\begin{lstlisting}[language=Go,firstnumber=108,basicstyle=\ttfamily\tiny,
  caption={\srcnote{middleware/circuitbreaker.go:108--128 report}}]
func (cb *circuitBreaker) report(failed bool) {
    switch circuitState(cb.state.Load()) {
    case stateClosed:
        if failed {
            if cb.failures.Add(1) >= cb.opt.FailureThreshold {
                cb.tripOpen()             // 连续失败到阈值
            }
        } else { cb.failures.Store(0) }   // 成功重置
    case stateHalfOpen:
        if failed { cb.tripOpen()         // 任一失败 -> Open
        } else if cb.halfOpenReq.Load() >= cb.opt.HalfOpenMaxReq {
            cb.state.Store(int32(stateClosed)) // 全成功 -> Closed
            cb.failures.Store(0)
        }
    }
}
\end{lstlisting}
\scriptsize ① Closed 失败计数；② 成功重置避免偶发抖动跳闸；③ HalfOpen \textbf{非对称}——任一失败立即回 Open，必须全成功才回 Closed；④ "信任成本"对资金链路合适：宁保守不再打死下游。
\end{frame}

\section{第四层：异步下单削峰}
% ============================================================

\begin{frame}{双 11 0 点：100k QPS 怎么扛}
\begin{itemize}
  \item 同步下单链路：HTTP $\to$ ReserveStock $\to$ INSERT order $\to$ COMMIT $\to$ 回客户端，p99 大约 50--100ms。
  \item 100k QPS $\times$ 100ms = \textbf{1 万并发}进 MySQL，连接池打爆。
  \item gomall 的方案：把"DB 落库"从同步链路里拆出来。
  \begin{enumerate}
    \item 同步部分：reserve 库存 + 写 ticket + 投 MQ $\to$ 返回 ticket；
    \item 异步部分：consumer 慢慢消费 MQ $\to$ INSERT order $\to$ 回写 ticket 状态；
    \item 客户端：拿 ticket 轮询 \texttt{/orders/status}。
  \end{enumerate}
  \item 收益：DB 压力从"瞬时尖刺"变成"恒定 1k QPS 平流"。
\end{itemize}
\srcnote{service/order\_async.go:119--172 OrderEnqueue}
\end{frame}

\section{第五层：HTTP cache 把读路径 CDN 化}
% ============================================================

\begin{frame}{读流量的最优解：根本不到 gomall}
\begin{itemize}
  \item P1 P3 读接口（\texttt{/product/show}, \texttt{/carousels}, \texttt{/category/list}）的数据更新频率\textbf{远低于查询频率}。
  \item 让浏览器 / CDN 缓存代答：
  \begin{itemize}
    \item 浏览器：直接返回 304 Not Modified（不传 body）；
    \item CDN：命中缓存（不到 gomall）。
  \end{itemize}
  \item gomall 实现：\texttt{HTTPCache(maxAge)} 中间件统一发 ETag + Cache-Control。
\end{itemize}
\srcnote{middleware/httpcache.go:47--100 HTTPCache; routes/router.go:39--45 挂载点}
\end{frame}

  \draw[arrow] (c) -- node[above,font=\tiny]{2.协商缓存} (cdn);
  \draw[arrow-async] (cdn) -- node[above,font=\tiny]{3.回源 ETag} (gw);
  \draw[arrow] (gw) -- (db);
\end{tikzpicture}
\vspace{2mm}
\begin{itemize}
  \item \textbf{强缓存}（Cache-Control: max-age=60）：浏览器本地命中，0 RTT；
  \item \textbf{协商缓存}（ETag + If-None-Match）：到 CDN，命中返回 304，无 body；
  \item \textbf{回源}：CDN 不命中，回 gomall；ETag 用 SHA-256 前 16 字节算。
\end{itemize}
\srcnote{middleware/httpcache.go:78--98 ETag 协商；routes/router.go:39 product/list max-age=30}
\end{frame}

\begin{frame}{各接口的 max-age 定多少}
\begin{tabular}{@{}lll@{}}
\toprule
接口 & max-age & 业务依据 \\
\midrule
/product/list   & 30s  & 价格 / 库存可能变化，不能太长 \\
/product/show   & 60s  & 商品详情更稳定 \\
/category/list  & 300s & 分类几乎不变 \\
/carousels      & 300s & 运营手动调，时效性弱 \\
\bottomrule

\small max-age 的\textbf{业务取舍}：太长会让"改价 / 下架"延迟生效，太短卸不了流量。\\
300s = 运营改完最多 5min 生效，可接受；30s = 库存接近灵敏阈值的折中。
\srcnote{routes/router.go:39--45 各路由的 maxAge 参数}
\end{frame}

\begin{frame}{error budget：用数字管"什么时候停发版"}
\begin{itemize}
  \item P0 SLO=99.95\% $\to$ 月度\textbf{宕机预算 $\approx$ 21min}。
  \item 烧 1/3（7min）减发版；烧 2/3（14min）冻结非紧急；全烧光 $\to$ 下月转"稳定性 only"。
  \item 预算用 SLI 推：限流命中率 / 熔断打开时间 / 5xx 比例累加成"不可用时间"。
\end{itemize}
\keypoint{流量治理不是\textbf{一个机制}，是\textbf{五个机制按业务等级组合}。\\
每个阈值都要能用业务语言解释清楚。}

\end{frame}

\begin{frame}{面试 Q\&A (1/2)}
\textbf{Q1}：100 RPS / 200 burst 这两个数字怎么定？\\
\hspace{4mm}\small A：基线 64k RPS / 单 IP 用不到 1/600，100 留给爬虫脚本做兜底；burst 200 容纳一次首屏并发 100+，不会误伤真用户。

\vspace{2mm}
\textbf{Q2}：为什么秒杀用 SlidingWindow 不用 TokenBucket？\\
\hspace{4mm}\small A：① 必须按用户而不是 IP，否则 NAT 出口全员共享；② 必须分布式精确，多实例下用 Redis 共享状态；③ ZSet 比 fixed-window 没有窗口边界可以利用。

\vspace{2mm}
\textbf{Q3}：CircuitBreaker 5/10s/3 三个数字有什么讲究？\\
\hspace{4mm}\small A：5 = 容忍偶发抖动不一抖就跳；10s = 给下游一次完整告警+自愈周期；3 = HalfOpen 探测够小，避免再次打死下游。
\end{frame}

\begin{frame}{面试 Q\&A (2/2)}
\textbf{Q4}：异步下单是不是会丢单？\\
\hspace{4mm}\small A：不会。库存先 Reserve（Redis Lua 原子），任一失败都 ReleaseReservation 回滚；ticket TTL=1h 兜底；MQ 用 RabbitMQ 持久化 + 手动 ack。

\vspace{2mm}
\textbf{Q5}：为什么熔断返回 200 不返回 503？\\
\hspace{4mm}\small A：app 端老网关不能优雅处理 5xx；统一走业务码（70002）= 客户端只用维护一套 status switch；监控不要按 4xx/5xx 算错误率，必须看 status。

\vspace{2mm}
\textbf{Q6}：HTTP cache 的 max-age 怎么定？\\
\hspace{4mm}\small A：取决于"改价 / 下架"的最大可接受延迟。商品详情 60s，分类 / 轮播 300s，列表 30s（库存敏感）。
\end{frame}
