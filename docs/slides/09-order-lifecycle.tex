\begin{frame}{从"消息能不能送达"理解状态机难点}
\begin{itemize}
  \item Created$\to$Paid：依赖\textbf{支付回调}。网络抖一下回调就丢，需要查询接口轮询。
  \item Paid$\to$Picking：依赖\textbf{WMS 拣货事件}。WMS 是另一套系统，得对账。
  \item Picking$\to$Shipped：\textbf{物流接单接口}。顺丰 / 京东物流各一套 webhook。
  \item Shipped$\to$Delivered：\textbf{快递公司签收回执}，延迟 0--72h，需 fallback 轮询。
\end{itemize}
\keypoint{核心难点\textbf{不是状态多}，而是每一次转移都跨系统、跨网络、跨组织边界。}
\srcnote{对照美团技术博客 2021《订单状态机的演进》}
\end{frame}

\section{30 分钟关单 + RMQ/Cron 双保险}
% ============================================================

\begin{frame}{30 分钟从哪定}
\begin{tabular}{@{}lll@{}}
\toprule
方案 & 用户感受 & 业务后果 \\
\midrule
5 分钟    & 切卡 / 等短信都不够   & 商家投诉"刚下单就被取消"，付款转化掉 10\%+ \\
15 分钟   & 多数人够用            & 高峰仍有 5\%+ 误关，弱网用户最先被牺牲 \\
\textbf{30 分钟} & 含切卡 + 短信冗余  & 大促锁单可控，商家 / 运营都能接受 \\
60 分钟   & 用户不抱怨            & 双 11 锁 1 小时，爆款一直显示"已售罄" \\
24 小时   & 客服零投诉            & 库存周转崩盘，恶意囤货成本为零 \\
\bottomrule
\end{tabular}
\vspace{2mm}
\small 商家投诉曲线和库存周转曲线是\textbf{反向的}，30 分钟是二者交点附近的工业经验值。\\
京东 30min / 天猫 30min--1h / 拼多多 15min —— 越是高频低价类目越短。
\srcnote{repository/rabbitmq/order\_delay.go:21--22 OrderCancelDelay = 30 * time.Minute}

  \item Redis ZSet 是早期实现，被 RMQ 替换后忘删，目前\textbf{只写不读}，等下一次清理 PR。
  \item 业务侧只关心 \textbf{用户实际看到 30 分钟}；技术侧三处数字的语义必须文档化。
\end{itemize}

\end{frame}

\begin{frame}{单一机制都不可靠}
\begin{tabular}{@{}lll@{}}
\toprule
方案 & 致命弱点 & 何时露馅 \\
\midrule
只用 RMQ 延迟队列  & MQ 宕机消息全丢      & 机房断网 / RabbitMQ 重启 \\
只用 Cron 5min    & 高峰锁单累积 5--15min & 双 11 库存周转跟不上 \\
只用 Redis ZSet   & Redis 内存满 evict   & 6M 历史订单一次性扫描 \\
\bottomrule
\end{tabular}
\vspace{2mm}
\begin{itemize}
  \item RMQ 是\textbf{时效快}（30min 准点到秒级）、\textbf{失败盲}（消息丢就是丢）。
  \item Cron 是\textbf{时效粗}（5min 步进）、\textbf{保证强}（DB 全表扫一定能扫到）。
  \item 两个机制\textbf{失败模式互不相关}，组合后整体可靠性 $\approx 1 - p_{rmq} \cdot p_{cron}$。
\end{itemize}
\srcnote{cmd/main.go:54--62 tryInitRabbitMQ 设计意图}
\end{frame}

\begin{frame}{为什么 outbox 必须在事务里}
\begin{tikzpicture}[node distance=6mm and 12mm,every node/.style={font=\scriptsize}]
  \node[node-box,fill=cMain!10]                       (db) {DB tx 提交\\order.type=Cancelled};
  \node[node-box,fill=cMain!10,right=of db]           (ob) {outbox 行\\order.cancelled};
  \node[node-box,fill=cWarn!15,right=of ob]           (pub) {Publisher\\异步抓取};
  \node[node-box,fill=cOK!10,right=of pub]            (down) {下游消费\\(订阅 / 推送 / BI)};
  \draw[arrow] (db) -- node[above,font=\tiny]{同事务} (ob);
  \draw[arrow-async] (ob) -- (pub);
  \draw[arrow] (pub) -- (down);
\end{tikzpicture}
\begin{itemize}
  \item 如果用 \texttt{tx.Commit() $\to$ MQ.Publish()} 两步，第二步失败会丢事件。
  \item Outbox 让事件\textbf{和业务数据同生死}：DB 提交即事件持久化，Publisher 异步搬运。
  \item 这就是 \texttt{order\_cancel.go:40--50} 把 \texttt{OutboxDao.Insert} 放在 \texttt{Transaction} 回调里的原因。
\end{itemize}
\srcnote{service/order\_cancel.go:40--50；配套 deck 04-outbox-saga.tex}
\end{frame}

\begin{frame}{售后链路：远比退款复杂}
\begin{tabular}{@{}ll@{}}
\toprule
售后类型 & 业务复杂度 \\
\midrule
仅退款 (Refund Only)        & 钱回去就行，库存不动 \\
退货退款 (Return + Refund)  & 用户寄回 + 入库质检 + 退钱 + 库存加回 \\
换货 (Exchange)             & 退货 + 新发货，资金不动 \\
价保 (Price Match)          & 不退货只退差额，营销系统介入 \\
仅维修 (Repair Only)        & 寄回 + 维修 + 寄回，跨 30--90 天 \\
\bottomrule
\end{tabular}

\begin{itemize}
  \item 五种售后\textbf{业务流程完全不同}，但都要在订单上记录"已发起售后单"状态。
  \item 大型电商通常用\textbf{售后单}另一张表，订单只标"有 N 个售后单关联"。
  \item gomall 当前没有任何售后表 —— 第一版能接通"仅退款"就是巨大进步。
\end{itemize}

\begin{frame}{面试 Q\&A}
\small
\textbf{Q1}：为什么不只用 RMQ，要再加 Cron？\\
\hspace{4mm} A：RMQ 宕机或消息丢就关不了单。Cron 在 DB 扫，失败模式互不相关，整体可用率 $\approx 1 - p_{rmq} \cdot p_{cron}$。

\vspace{1mm}
\textbf{Q2}：\texttt{CancelUnpaidOrder} 怎么保证幂等？\\
\hspace{4mm} A：\texttt{UPDATE WHERE type=UnPaid} 乐观锁，\texttt{RowsAffected=0} 时 no-op，不释放库存不发 outbox。四调用方共用这把锁。

\vspace{1mm}
\textbf{Q3}：\texttt{"* */5 * * * *"} 为什么是 bug？\\
\hspace{4mm} A：6 段格式第一段是秒，\texttt{*} 每秒都跑。在分钟 \% 5==0 时 5min 跑 60 次。正确 \texttt{"0 */5 * * * *"}。

\vspace{1mm}
\textbf{Q4}：状态机为什么只有 3 态，缺了什么？\\
\hspace{4mm} A：缺 Picking / Shipped / Delivered / Refunded —— 每态都要跨系统对接 WMS / 物流 / 快递 / 资金接口。
\end{frame}
