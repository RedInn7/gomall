\begin{frame}{雪花 64-bit 结构示意}
\begin{tikzpicture}[x=8mm,y=8mm]
  % 整条 8 段，每段代表若干位
  \draw[draw=cMain,thick] (0,0) rectangle (1,1);

  \draw[draw=cAccent,thick,fill=cAccent!18] (6,0) rectangle (8,1);

    \node[font=\tiny,below] at (9,0)    {12 bit};
\end{tikzpicture}
\vspace{4mm}
\begin{itemize}
  \item \textbf{41 bit 时间戳}：从 2024-01-01 起，可用 $\approx$ 69 年。
  \item \textbf{10 bit 机器位}：最多 1024 个进程，gomall 单实例够用。
  \item \textbf{12 bit 序列号}：单进程单毫秒最多 4096 个 ID。
\end{itemize}
\srcnote{pkg/utils/snowflake/snowflake.go:11 epoch = 2024-01-01 UTC}
\end{frame}

\begin{frame}[fragile]{InitSnowflake：epoch 与机器位}
\begin{lstlisting}[language=Go,firstnumber=10,
  caption={\srcnote{pkg/utils/snowflake/snowflake.go:10--21}}]
func InitSnowflake(machineID int64) {
    snowflake.Epoch = time.Date(
        2024, 1, 1, 0, 0, 0, 0, time.UTC,
    ).UnixNano() / 1000000
    var err error
    node, err = snowflake.NewNode(machineID)
    if err != nil {
        panic(err)
    }
}

func GenSnowflakeID() int64 {
    return node.Generate().Int64()
}
\end{lstlisting}
\end{frame}

\begin{frame}{逐行讲：为什么这 12 行很要紧}
\begin{description}\setlength{\itemsep}{0.4mm}
  \item[\texttt{Epoch=2024-01-01}] 把基准从 1970 平移到 2024，腾出最高位的时间空间，41 bit 能多撑几十年。
  \item[\texttt{machineID}] 部署时由配置注入，多副本必须各不相同；冲突 = 双发出现重复订单号。
  \item[\texttt{NewNode 失败 panic}] 启动期就挂掉，比运行期偶发生成失败更安全 —— 订单号是金融语义，不能"先跑起来再说"。
  \item[\texttt{Generate().Int64()}] 一次返回 int64，方便存 \texttt{order.order\_num bigint unsigned}。
\end{description}
\vspace{2mm}
\keypoint{雪花是\textbf{本地无锁}的——OrderCreate 调用它零网络成本，这是 50K RPS 的前提。}
\end{frame}

\begin{frame}[fragile]{OrderCreate 主体三步}
\begin{lstlisting}[language=Go,firstnumber=46,
  caption={\srcnote{service/order.go:46--86}}]
order := &model.Order{
    UserID:    u.Id,
    ProductID: req.ProductID,
    Num:       int(req.Num),
    OrderNum:  uint64(snowflake.GenSnowflakeID()),
}

// 1) Redis 预扣库存（事务外）

    return nil, err
}

// 2) 同事务写订单 + 写 outbox
err = dao.NewDBClient(ctx).Transaction(func(tx *gorm.DB) error {
    if e := dao.NewOrderDaoByDB(tx).CreateOrder(order); e != nil {
        return e
    }
    return dao.NewOutboxDaoByDB(tx).Insert(
        "order", "OrderCreated", "order.created", order.ID,
        events.OrderCreated{ OrderID: order.ID, OrderNum: order.OrderNum,
            UserID: u.Id, ProductID: req.ProductID, Num: int(req.Num) })
})
if err != nil {
    // 3) 事务失败：把刚扣的预占库存放回
    _ = cache.ReleaseReservation(ctx, req.ProductID, int64(req.Num))
}

\begin{frame}{逐行讲：为什么 reserve 在 tx 外}
\begin{itemize}
  \item \textbf{Redis 不能加入 MySQL 事务}：跨存储两阶段提交业内基本不做，复杂度收益不划算。
  \item \textbf{reserve 失败必须早退}：库存不足 = 业务码 \texttt{50001}，不该让 DB 也写一条又回滚。
  \item \textbf{tx 失败要补偿 release}：如果 reserve 成功但 DB 失败，必须把扣下的库存放回 available，否则虚假占用。
  \item \textbf{补偿不是事务的一部分}：即便 \texttt{ReleaseReservation} 也失败，DB 已 rollback，库存差额由对账兜底（\texttt{GetStockSnapshot} 巡检）。
\end{itemize}
\srcnote{repository/cache/inventory.go:78--125 reserve / commit / release 三脚}
\end{frame}

\begin{frame}{逐行讲：为什么 outbox 必须在 tx 内}
\begin{itemize}
  \item 写订单 + 发 MQ 如果\textbf{拆成两步}：
  \begin{itemize}
    \item 写订单成功、发 MQ 前进程挂了 → \textbf{下游永远收不到 OrderCreated}。
    \item 反过来：先发 MQ 再写 DB，DB 失败 → \textbf{幻影订单事件}。
  \end{itemize}
  \item \textbf{outbox 表充当桥梁}：写订单和写 outbox 用同一 \texttt{INSERT} 事务，要么都成功要么都没有。
  \item \textbf{异步 publisher} 扫 \texttt{status=pending} 的 outbox 行，发 MQ，发完更新成 \texttt{sent}。
  \item 这就是 deck 04 (旧) Outbox 模式在 deck 04 (新) 的落地点。
\end{itemize}
\srcnote{repository/db/dao/outbox.go:26--41 Insert 同 tx；service/outbox/publisher.go 扫表 publisher}
\end{frame}

func (s *OrderSrv) OrderEnqueue(ctx context.Context,
    req *types.OrderCreateReq) (resp interface{}, err error) {

    if err = cache.ReserveStock(ctx,
        req.ProductID, int64(req.Num)); err != nil {
        return nil, err   // 50001 库存不足直接拒
    }

    ticket := fmt.Sprintf("%d", snowflake.GenSnowflakeID())

\begin{frame}{逐行讲：ticket 与 TTL 的取舍}
\begin{itemize}
  \item \textbf{ticket 用雪花生成}：和订单号同源，全局唯一、客户端不能伪造。
  \item \textbf{TTL = 1h}：覆盖最长支付窗口（15min）+ 客服查单 + 一次客户端崩溃。
  \item \textbf{ticket 不等于订单号}：consumer 落库后回写 \texttt{OrderTicketStatus.OrderNum}，客户端读 status 时一并拿到。
  \item \textbf{失败也写 ticket}：投递失败回写 \texttt{status=failed, reason=...}，前端不用纠结超时还是真失败。
\end{itemize}
\srcnote{service/order\_async.go:23--52 常量与状态结构}
\end{frame}

\begin{frame}{对账：reserve / commit / release 不变量}
\begin{itemize}
  \item 库存 Redis 双桶：\texttt{available + reserved + sold = 初始水位}。
  \item DB 订单 + outbox 同事务：两条记录要么都在要么都没有。
  \item ticket + DB 行：\texttt{OrderTicketStatus.Status=ok} $\iff$ \texttt{orders.order\_num} 存在。
\end{itemize}
\vspace{2mm}
任何一个不变量被破坏，巡检脚本应当告警：
\begin{itemize}
  \item Redis 双桶不平 → 库存对账脚本（\texttt{GetStockSnapshot}）。
  \item 订单与 outbox 数量不一致 → 直接 SQL JOIN 抽查。
  \item ticket=ok 但订单不存在 → consumer 写回路径异常。
\end{itemize}
\srcnote{repository/cache/inventory.go:128--138 GetStockSnapshot 巡检入口}
\end{frame}

\begin{frame}{逐行讲：CAS 防一券多用}
\begin{description}\setlength{\itemsep}{0.5mm}
  \item[\texttt{status = Unused}] WHERE 加状态条件 = CAS：两个并发请求只有一个能把 1 改成 locked。
  \item[\texttt{RowsAffected==0}] 不是"找不到"，而是\textbf{已经被别人锁了}，必须返回业务码而不是 500。
  \item[\texttt{used\_order\_id=0 占位}] 同 tx 内此时 \texttt{order.ID} 还没生成，先占位 0，待 \texttt{CreateOrder} 后回填。
  \item[\texttt{recalcOrderMoney}] 校验 \texttt{order.money >= batch.threshold}，否则券不可用——\textbf{后端校验}比前端报价权威。
  \item[\texttt{writeOrderAndOutbox}] 上面三步任何一步失败都连同订单一起 rollback，券状态自动回 1。
\end{description}

\begin{frame}{逐行讲：为什么要 ParentNum + 部分回滚}
\begin{itemize}
  \item \textbf{ParentNum}：用户视角"一次下单"的逻辑标识；客服查单按它聚合，分账按它对账。
  \item \textbf{逐组 tx 而非一个大 tx}：N 个店铺事务串行，单 tx 大事务会锁太多行 + 跨表锁，吞吐量塌方。

  \item \textbf{outbox 自然多条}：N 个子订单 = N 条 OrderCreated 事件，下游 (发货、对账) 各按子订单消费。
\end{itemize}

\begin{frame}{为什么 epoch 选 2024-01-01 而不是 1970}
\begin{itemize}
  \item 标准 unix 时间戳 \texttt{1970-01-01} 到 2024 已经过去 $\approx$ 54 年 $\approx 1.7 \times 10^{12}$ ms。
  \item 41 bit 时间戳上限 $\approx 2.2 \times 10^{12}$ ms $\approx$ 69.7 年。
  \item 如果 epoch 用 1970：\textbf{剩余可用 $\approx$ 16 年}，2040 年雪花溢出。
  \item 如果 epoch 用 2024：\textbf{剩余可用 $\approx$ 69 年}，到 2094 年都不用动。
\end{itemize}
\vspace{2mm}
\begin{tikzpicture}[x=12mm,y=4mm]
  \draw[draw=cGray,thick] (0,0) -- (8,0);
  \foreach \x/\l in {0/1970,1.6/2024,4.5/2040,8/2094} {

\end{frame}

\begin{frame}{时钟回拨：业内三种处理方式}
\begin{itemize}
  \item \textbf{阻塞等待}：检测到 \texttt{now < lastTimestamp} 时，sleep 到追上为止。库内置（bwmarrin/snowflake 默认行为）。
  \item \textbf{扩展位补偿}：保留几个 bit 做"回拨计数器"，发生回拨时 +1，时间戳继续走。Twitter 原版。
  \item \textbf{停机告警}：差距 > 阈值（如 5s）直接拒绝 ID 生成，让 SRE 介入。金融级常用。
\end{itemize}
\vspace{2mm}
\begin{itemize}
  \item gomall 走第一种 —— \texttt{NewNode 失败 panic} 保证启动健康，运行期由库阻塞。
  \item 配合 NTP \texttt{slew} 模式（chronyd 默认）让时钟\textbf{平滑追赶}而不是跳变。
  \item 监控锚点：\texttt{GenSnowflakeID p99} —— 突然变大就是回拨告警的前哨。
\end{itemize}
\srcnote{pkg/utils/snowflake/snowflake.go:13--16 panic 策略；NTP slew vs step 由运维保证}
\end{frame}

\begin{frame}{面试 Q\&A（1/3）}
\small
\textbf{Q1}：为什么 reserve 不放进 DB 事务？\\
\hspace{4mm}A：跨 Redis + MySQL 两阶段提交收益不抵复杂度。失败由 ReleaseReservation 补偿，对账兜底。
\vspace{1mm}

\textbf{Q2}：outbox 表写满了怎么办？\\
\hspace{4mm}A：publisher 周期归档 sent；dead 进单独表人工处理。归档脚本按月按 status 分区清。
\vspace{1mm}

\textbf{Q3}：雪花时钟回拨在 gomall 怎么处理？\\
\hspace{4mm}A：库内部阻塞等待，单机部署 NTP 用 slew。监控 GenSnowflakeID p99 作为预警。
\vspace{1mm}

\textbf{Q4}：50001 和 60002 客户端怎么区分对待？\\
\hspace{4mm}A：50001 终止重试、提示售罄；60002 用旧响应跳订单详情，不重发。
\end{frame}

\begin{frame}{面试 Q\&A（2/3）}
\small
\textbf{Q5}：/orders/enqueue 比 /orders/create 多了什么风险？\\
\hspace{4mm}A：多一个 ticket 状态需要轮询；consumer 不可用时 ticket 一直 pending 需要 TTL 兜底。
\vspace{1mm}

\textbf{Q6}：跨店满减要做，购物车要改什么？\\
\hspace{4mm}A：聚合算价从 user 维度改为 (user, boss\_id) 维度，下单也要拆成多店多单。
\vspace{1mm}

\textbf{Q7}：平台券 + 店铺券同时挂，怎么防互斥规则被绕过？\\
\hspace{4mm}A：服务端按 \texttt{batch.scope} 二次校验，前端禁用只是 UX；下单 tx 内 CAS 锁券。
\vspace{1mm}

\textbf{Q8}：拆单第 3 个店铺事务失败，前两个怎么办？\\
\hspace{4mm}A：父订单整体失败，前两个落库的子订单走"业务侧回滚"——ReleaseReservation + 标 closed。
\end{frame}

\begin{frame}{面试 Q\&A（3/3）}
\small
\textbf{Q9}：雪花 epoch 选 2024-01-01 之后想再改怎么办？\\
\hspace{4mm}A：不能动。同表里两种 epoch 解出的 ID 会撞，必须新表新字段或保持 epoch 永久不变。
\vspace{1mm}

\textbf{Q10}：100k QPS 营销活动，同步还是异步？\\
\hspace{4mm}A：异步 + 排队页 + 限流。同步路径留给 1k 量级的"普通下单"。
\vspace{1mm}

\textbf{Q11}：outbox dead 行如何避免越积越多？\\
\hspace{4mm}A：告警 + 归档 + 人工补单出口；超过 7 天转 outbox\_archive 表。
