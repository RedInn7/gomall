\begin{frame}{先认人：一笔订单背后站着 5 个角色}
\begin{itemize}
  \item \textbf{C 端用户}：进门 $\to$ 逛 $\to$ 下单 $\to$ 收货。只关心\textbf{快、对、不丢}。
  \item \textbf{商家（Boss）}：订单是"我能不能发货"的唯一源头；货卖出去钱要到账。
  \item \textbf{运营}：办活动拉量 —— 优惠券 / 满减 / 秒杀 / 拼团 / 预售，预算不能被刷穿。
  \item \textbf{客服}：用户来问"我的钱去哪了 / 货到哪了"，得有一条能查的状态线。
  \item \textbf{SRE}：大促洪峰别把库存卖超、别把数据库打挂，挂了要能降级。
\end{itemize}
\vspace{2mm}
\keypoint{这套 deck 按\textbf{业务域}垂直切成 15 份，每份讲透一个域。\\
本份是横向的"地图"：把 15 个域串成用户真实走过的一条线。}

\begin{frame}{业务承诺：我们对外兜什么底（SLO）}
  \keypoint{一个交易系统的可信度，不在于功能多，而在于敢把哪几条线写进合同——出了问题要赔的那种。gomall 把承诺分成 P0\textasciitilde P3 四档，资源向核心交易倾斜。}
  \vspace{0.4em}
  \begin{table}
  \centering\small
  \begin{tabular}{@{}llcc@{}}
  \toprule
  \textbf{等级} & \textbf{典型场景} & \textbf{可用率} & \textbf{p99 延迟} \\
  \midrule
  \textcolor{cBad}{\textbf{P0 核心交易}}  & 下单 / 支付       & \textbf{99.95\%} & < 500ms \\
  \textcolor{cAccent}{P1 核心读}          & 商品详情 / 订单列表 & 99.9\%  & < 200ms \\
  \textcolor{cWarn}{P2 营销秒杀}          & 抢券 / 秒杀 / 红包  & 99\%    & < 200ms \\
  \textcolor{cGray}{P3 信息类}            & 轮播 / 分类         & 99\%    & < 1s \\
  \bottomrule
  \end{tabular}
  \end{table}
  \vspace{0.3em}
  \begin{itemize}\small
    \item \textbf{钱货不丢}：支付与库存、资金流水落同一事务，崩了整笔回滚，绝不出现"扣了钱没货"。
    \item \textbf{happy path 不被周边阻断}：优惠 / 搜索 / 通知挂了，下单支付照走。
    \item \textbf{宁可下游慢、不能上游 429}：P0 接口（下单 / 支付）靠缓存 + 异步削峰扛峰值。
  \end{itemize}
  \srcnote{README.md \S 业务承诺（SLO）}
\end{frame}

\begin{frame}[fragile]{钱怎么流：资金台账与复式记账}
  \keypoint{一笔支付从来不是"给买家余额减个数"。买家少的钱必须有人多——商家。复式记账让每一分钱都有去向，对账时 SUM(debit) 永远等于 SUM(credit)。}
  \vspace{0.2em}
  \begin{lstlisting}[language=Go,caption={同事务内一借一贷，方向相反}]
// 买家扣款流水：与余额变动同事务追加
ledgerDao := money.NewLedgerDaoByDB(tx)   // 接管调用方 tx
ledgerDao.AppendTransaction(uId, order.ID,
    money.DirectionDebit, payable, buyerBalanceAfter, money.BizTypeOrderPay)
// 卖家入账流水：方向与买家相反，金额相同
ledgerDao.AppendTransaction(bossID, order.ID,
    money.DirectionCredit, payable, bossBalanceAfter, money.BizTypeOrderPay)
  \end{lstlisting}
  \vspace{-0.2em}
  \begin{itemize}\small
    \item \textbf{同事务}：\texttt{NewLedgerDaoByDB(tx)} 复用余额变动那把锁的同一 tx，流水与余额原子提交。
    \item \textbf{防重账}：\texttt{(ref\_order\_id, direction, biz\_type)} 唯一索引兜底，重放回调撞唯一键报错 $\to$ 整事务回滚。
    \item \textbf{外部资金}：Stripe / Web3 买家不走内部钱包，debit 对手方记在清算账户（user\_id=0），守恒不破。
  \end{itemize}
  \srcnote{internal/payment/service.go:122 一借一贷；internal/money/ledger\_repo.go AppendTransaction；ledger\_model.go 唯一索引}
\end{frame}

  \keypoint{选型不是堆时髦框架，而是"这个痛点用什么扛最划算"。}
  \vspace{0.2em}
  \begin{table}
  \centering\footnotesize
  \begin{tabular}{@{}lll@{}}
  \toprule
  \textbf{组件} & \textbf{职责} & \textbf{为什么用它} \\

\begin{frame}{中间件链：逐环拆解，顺序即设计}
  \keypoint{中间件的顺序不是随手排的——每一环只在前一环放行后才有意义，错一步就是漏洞。}
  \vspace{0.3em}
  \begin{center}
  \begin{tikzpicture}[node distance=0.42cm and 0.5cm]
    \node[node-box] (jwt) {JWT 鉴权};
    \node[node-box, right=of jwt] (rbac) {RBAC};
    \node[node-box, right=of rbac] (rate) {限流};
    \node[node-box, right=of rate] (cb) {熔断};
    \node[node-box, right=of cb] (idem) {幂等};
    \draw[arrow] (jwt) -- (rbac);
    \draw[arrow] (rbac) -- (rate);
    \draw[arrow] (rate) -- (cb);
    \draw[arrow] (cb) -- (idem);
  \end{tikzpicture}
  \end{center}
  \vspace{0.2em}
  \begin{itemize}\small
    \item \textbf{JWT 鉴权} \texttt{AuthMiddleware()}：解 token 拿 user\_id，挂 \texttt{authed} 组，没登录直接 401（全局：令牌桶限流 + 链路追踪先于一切）。
    \item \textbf{RBAC} \texttt{RequireRole("admin")}：登录之上叠角色校验，只护 admin 组。\textbf{鉴权必须在限流前}——否则匿名洪水也吃配额。
    \item \textbf{限流 + 熔断}：全局 \texttt{TokenBucket(100,200)} 防爬虫，秒杀路由再叠滑动窗口；支付路由专用熔断，下游连错 5 次 open 10s。
    \item \textbf{幂等} \texttt{Idempotency()}：\textbf{放最后}——确认要执行业务时才用 Redis 锁去重，避免给被拒请求白占幂等键。
  \end{itemize}
  \srcnote{routes/router.go:41 全局链；middleware/\{jwt,rbac,ratelimit,circuitbreaker,idempotency\}.go；internal/payment/routes.go}
\end{frame}

\begin{frame}{数据层选型与边界：一个 MySQL 为什么不够}
  \keypoint{把全部活儿压给一个 MySQL，要么慢死、要么挂掉。正解按"访问形态"分流：强一致归 MySQL，高并发与检索归旁路，旁路挂了能退回 DB。}
  \vspace{0.2em}
  \begin{table}
  \centering\small
  \begin{tabular}{@{}lll@{}}
  \toprule
  \textbf{存储} & \textbf{扛什么} & \textbf{挂了怎么办} \\
  \midrule
  MySQL 主从     & 交易事务、权威读             & 核心依赖，不可降级 \\
  Redis          & 库存预扣 / 热点缓存 / 限流     & 缓存未命中回源 DB \\
  Elasticsearch  & 商品全文检索                 & \textbf{退回 DB \texttt{LIKE} 查询} \\
  Milvus         & 语义向量召回                 & 退回纯 ES 关键词召回 \\
  \bottomrule
  \end{tabular}
  \end{table}
  \vspace{0.2em}
  \begin{itemize}\small
    \item \textbf{读写分离}：\texttt{dbresolver} 写走 Sources、读走 2 个 Replicas，\texttt{RandomPolicy} 负载均衡。
    \item \textbf{搜索降级}：ES 不可用时落到 DB \texttt{name/info LIKE}；语义召回报错则退回纯向量——周边能力只增强体验，绝不阻断 happy path。
  \end{itemize}

\begin{frame}{黄金路径：7 跳，每一跳都可能掉链子}
\begin{enumerate}
  \item \textbf{登录} —— 拿到 JWT，后续每个请求带着它过鉴权中间件。
  \item \textbf{看商品} —— 详情走 Cache Aside，热点不打穿 DB。

\begin{frame}{满减最优解：多规则叠加时帮用户算到最便宜}
\keypoint{用户身上可能同时命中好几条满减规则，系统帮他挑出"减得最多"的那一条；算折扣失败就降级为无折扣，绝不阻断下单。}
\begin{itemize}\small
  \item \textbf{先算后下单，两段分工}：结算/购物车实时展示走 CalculateBestDiscount（纯只读、不落库）；真正下单时在事务内 ApplyDiscountInTx 扣减规则预算并落账。
  \item \textbf{怎么算"最优"}：PickBestPromoRule 遍历所有命中规则，逐条算可减金额，取最大的一条；没有命中或减额 $\le 0$ 就返回原价。
  \item \textbf{失败降级}：CalculateBestDiscount 报错时记 warn 日志、回退无折扣继续下单；规则畸形导致折扣 > 小计时，兜底把实付夹到 0，不让用户实付为负。
\end{itemize}
\vspace{1mm}
\keypoint{两段分工一句话：\textbf{结算页只读算最优（CalculateBestDiscount，不落库），下单事务内才落账（ApplyDiscountInTx，超支回滚）}；任一异常都降级为无折扣、绝不阻断成交。}

\end{frame}

\begin{frame}[fragile]{硬骨头 1：库存防超卖 —— 为什么不用 \texttt{UPDATE WHERE stock>0}}
\begin{itemize}
  \item 高并发下行锁排队 + 尾延迟爆炸；下单链路最怕在事务里等锁。
  \item gomall 把库存搬到 Redis，用 \textbf{两个桶}：\texttt{available}（可卖）/ \texttt{reserved}（已占）。
  \item 预扣 / 提交 / 释放都是 \textbf{单个 Lua 脚本原子完成}，一次 RTT，天然不超卖。
\end{itemize}
\begin{lstlisting}[language=LuaRedis,caption={reserveScript：available 与 reserved 两个 key 原子流转}]

local avail = redis.call('GET', KEYS[1])
if avail == false then return -2 end           -- 库存未初始化
if tonumber(avail) < tonumber(ARGV[1]) then
    return -1                                  -- 不够，拒单
end
redis.call('DECRBY', KEYS[1], ARGV[1])         -- available -= n
redis.call('INCRBY', KEYS[2], ARGV[1])         -- reserved  += n
return 1

\begin{frame}{面试 Q\&A（1/2）}
\small
\textbf{Q1}：为什么库存预扣放 Redis 而不放 DB 事务里？\\
\hspace{4mm}A：DB 行锁在高并发下排队、尾延迟爆炸；Redis 两桶 Lua 一次 RTT 原子完成，下单链路不被锁拖住。失败由 ReleaseReservation 补偿 + 对账兜底。
\vspace{1mm}

\textbf{Q2}：下单为什么要 outbox，不直接发 MQ？\\
\hspace{4mm}A：直接发会有双写问题 —— 订单落库与消息投递不在一个事务，中间崩溃就丢事件。outbox 把事件写进同一个 DB 事务，再由 publisher 至少一次投递。
\vspace{1mm}

\textbf{Q3}：满减算挂了为什么不直接报错？\\
\hspace{4mm}A：优惠是锦上添花，不是下单前提。SLO 承诺保护 happy path，所以算失败、预算耗尽都降级为无折扣，用户照样能成交。
\end{frame}

\begin{frame}{面试 Q\&A（2/2）}
\small
\textbf{Q4}：事务提交后才发 MQ 延迟关单，万一 MQ 挂了订单永远不关？\\
\hspace{4mm}A：双保险。RabbitMQ 延迟队列（TTL+DLX）为主，DB Cron（每 5min 扫 WaitPay 超时单）兜底，MQ 不可用也能关单释放库存；两路共用幂等关单，重复触发只生效一次。
\vspace{1mm}

\textbf{Q5}：支付成功后库存是怎么真正减少的？\\
\hspace{4mm}A：下单只 reserved（预占）；支付事务改完状态后同步 CommitReservation，把 reserved 真正消掉。未支付被关单则 ReleaseReservation 还回 available。
\vspace{1mm}

\textbf{Q6}：游标分页相比深分页到底强多少？\\
\hspace{4mm}A：压测里订单列表游标+缓存 58,406 RPS / p95 5ms，旧版深分页 8.3 RPS / p95 15.95s —— 吞吐约 7,000$\times$、p95 约 3,200$\times$。
\end{frame}

\begin{frame}{面试 Q\&A（3/3）：一致性 · 熔断 · 检索}
\small
\textbf{Q7}：MySQL 主从读写分离下，读一致性怎么保证？\\
\hspace{4mm}A：普通读走从库分摊负载；对一致性敏感的资金/库存路径全部在事务内 \texttt{FOR UPDATE}（\texttt{LockTwoUsersForUpdate} / \texttt{GetUserByIdForUpdate}），事务与行锁读由 dbresolver 强制路由到主库，绕开从库延迟。
\vspace{1mm}

\textbf{Q8}：熔断器半开态怎么探测下游恢复？\\
\hspace{4mm}A：Open 超过 OpenTimeout（默认 10s）转半开，只放 HalfOpenMaxReq（默认 3）个探测；连续成功达阈值回 Closed，任一失败立刻回 Open。代际（generation）机制丢弃上一轮迟到回报，不误触发刚恢复的熔断器。
\vspace{1mm}

\textbf{Q9}：已有 ES，为什么还要 Milvus 向量召回？\\
\hspace{4mm}A：ES 擅长关键词精确匹配，"显瘦的连衣裙"这类语义查询命中差；Milvus 走 embedding 语义召回，双路召回后加权融合（min-max 归一）。ES 挂了降级走纯向量，不让语义检索整体不可用。
\end{frame}

\begin{frame}{面试 Q\&A（4/4）：投递去重 · ID 防撞 · 资金对账}
\small
\textbf{Q10}：outbox 至少一次投递，下游怎么防重复消费？\\
\hspace{4mm}A：不追求"恰好一次"，而是至少一次投递 + 消费端幂等吸收重复。落地手段是业务台账唯一索引——满减退预算 \texttt{promo\_release(order\_id)}、退款台账 \texttt{(ref\_order\_id, direction, biz\_type)}，重复投递撞键即幂等返回。
\vspace{1mm}

\textbf{Q11}：雪花 ID 多副本部署怎么防 ID 撞车？\\
\hspace{4mm}A：每副本必须配 \texttt{SNOWFLAKE\_NODE\_ID}（或 \texttt{NODE\_ID}），范围 \texttt{[0,1023]}；启动强校验——未配置或越界直接 \texttt{Fatal} 拒绝启动，仅 \texttt{SNOWFLAKE\_ALLOW\_DEFAULT=true} 时退回 node 0（限本地开发）。
\vspace{1mm}

\textbf{Q12}：支付为什么要复式记账？解决什么对账问题？\\
\hspace{4mm}A：余额密文落库不可对账；每次变动追加一条不可变明文流水，一次转移成对出现（买家 debit / 卖家 credit），恒等式 \texttt{SUM(debit)=SUM(credit)}。外部入口（Stripe/Web3）对手方记清算账户（user\_id=0），守恒不破，可逐笔追溯资金来龙去脉。
\end{frame}
