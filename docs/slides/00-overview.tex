\section{业务全景：gomall 到底在做什么}

\begin{frame}{gomall 是什么：一个讲清"为什么这么选"的电商后端}
\keypoint{用 \textbf{Go + Gin} 写的电商后端，把一笔订单从\textbf{浏览 $\to$ 搜索 $\to$ 下单 $\to$ 支付 $\to$ 履约}的全链路跑通；再叠上优惠券 / 秒杀 / 红包 / 拼团 / 预售、Web3 支付、ES+Milvus 向量检索。}
\vspace{0.3em}
\begin{columns}[T]
  \column{0.53\textwidth}
    \textbf{它到底在解决什么}
    \begin{itemize}\small
      \item \textbf{解决真实问题}：盯住真实电商里"该怎么选"的工程决策——超卖、一致性、限流、降级如何权衡。
      \item \textbf{20 个业务域}按 DDD 垂直切片，每域 \texttt{internal/<域>/} 五件套内聚，跨域写经属主服务落库。
      \item \textbf{4 条支付通道}（钱包 / Stripe / USDC / ETH）殊途同归，统一汇到\textbf{复式记账台账}，借贷守恒可对账。
      \item 每个域配\textbf{压测报告 + Beamer deck + 博客}，把当初的取舍也一并写下来。
    \end{itemize}
  \column{0.47\textwidth}
    \textbf{技术栈一览}
    \begin{itemize}\small
      \item \textbf{接入}：Gin + 中间件链（限流 / JWT / RBAC / 熔断 / 幂等 / Jaeger）
      \item \textbf{存储}：MySQL 主从 · Redis · Elasticsearch · Milvus
      \item \textbf{异步}：RabbitMQ + Outbox 旁路投递
      \item \textbf{链上}：Web3 Escrow + EVM 对账
      \item \textbf{可观测}：Jaeger 链路 + SkyWalking
    \end{itemize}
\end{columns}

\begin{frame}{先认人：一笔订单背后站着 5 个角色}
\begin{itemize}
  \item \textbf{C 端用户}：进门 $\to$ 逛 $\to$ 下单 $\to$ 收货。只关心\textbf{快、对、不丢}。
  \item \textbf{商家（Boss）}：订单是"我能不能发货"的唯一源头；货卖出去钱要到账。
  \item \textbf{运营}：办活动拉量 —— 优惠券 / 满减 / 秒杀 / 拼团 / 预售，预算不能被刷穿。
  \item \textbf{客服}：用户来问"我的钱去哪了 / 货到哪了"，得有一条能查的状态线。
  \item \textbf{SRE}：大促洪峰别把库存卖超、别把数据库打挂，挂了要能降级。
\end{itemize}
\vspace{2mm}

再逐段看每一跳背后的技术取舍，以及"出事会卡在哪"。}

\begin{frame}{中间件链：顺序即设计（为什么是这个次序）}
  \keypoint{顺序不是随手排的——每一环只在前一环放行后才有意义，错一步就是漏洞或浪费。下面按 \texttt{router.go} 真实注册顺序逐环讲。}
  \vspace{0.2em}
  \begin{center}
  \begin{tikzpicture}[node distance=0.42cm and 0.5cm]
    \node[node-box] (rate) {限流·IP};
    \node[node-box, right=of rate] (jwt) {JWT 鉴权};
    \node[node-box, right=of jwt] (rbac) {RBAC};
    \node[node-box, right=of rbac] (cb) {熔断};
    \node[node-box, right=of cb] (idem) {幂等};
    \draw[arrow] (rate) -- (jwt);
    \draw[arrow] (jwt) -- (rbac);
    \draw[arrow] (rbac) -- (cb);
    \draw[arrow] (cb) -- (idem);
  \end{tikzpicture}
  \end{center}
  \begin{itemize}\footnotesize
    \item \textbf{限流·IP} \texttt{TokenBucket(100,200)}：放\textbf{最外层}——匿名洪水在最便宜处被拒，不浪费后面的解析；被拒 HTTP 200 + 限流业务码。
    \item \textbf{JWT 鉴权} \texttt{AuthMiddleware()}：解 token 拿 user\_id，挂 \texttt{authed} 组；失败 HTTP 200 + 鉴权码（项目统一用 body status，不用 401）。
    \item \textbf{RBAC} \texttt{RequireRole("admin")}：登录之上叠角色校验，只护 admin 组——\textbf{必须先认证再鉴权}。
    \item \textbf{熔断}：支付等路由对下游调用加熔断，连错 5 次 open 10s，快速失败不被垂死依赖拖垮。
    \item \textbf{幂等} \texttt{Idempotency()}：\textbf{放最后}——确认要执行业务时才用 Redis 锁去重，避免给被拒请求白占幂等键。
    \item \textbf{注}：按\textbf{用户}维度的秒杀滑窗限流在鉴权\textbf{之后}（要先知道是谁），全局按 IP 的令牌桶才放最前。
  \end{itemize}
  \srcnote{routes/router.go:41--55 全局链；middleware/\{ratelimit,jwt,rbac,circuitbreaker,idempotency\}.go}
\end{frame}

\begin{frame}{数据层选型与边界：一个 MySQL 为什么不够}
  \keypoint{把全部活儿压给一个 MySQL，要么慢死、要么挂掉。正解按"访问形态"分流：强一致归 MySQL，高并发与检索归旁路，旁路挂了能退回 DB。}
  \vspace{0.2em}
  \begin{table}
  \centering\small
  \begin{tabular}{@{}lll@{}}
  \toprule
  \textbf{存储} & \textbf{扛什么} & \textbf{挂了怎么办} \\
  \midrule
  MySQL 主从     & 交易事务、权威读             & 核心依赖，不可降级 \\
  Redis          & 库存预扣 / 热点缓存 / 限流     & 缓存未命中回源 DB \\
  Elasticsearch  & 商品全文检索                 & \textbf{退回 DB \texttt{LIKE} 查询} \\
  Milvus         & 语义向量召回                 & 退回纯 ES 关键词召回 \\
  \bottomrule
  \end{tabular}
  \end{table}
  \vspace{0.2em}
  \begin{itemize}\small
    \item \textbf{读写分离}：\texttt{dbresolver} 写走 Sources、读走 2 个 Replicas，\texttt{RandomPolicy} 负载均衡。
    \item \textbf{搜索降级}：ES 不可用时落到 DB \texttt{name/info LIKE}；语义召回报错则退回纯向量——周边能力只增强体验，绝不阻断 happy path。
  \end{itemize}
  \srcnote{repository/db/dao/init.go dbresolver；service/search/ 降级路径；internal/product/repo.go LIKE}
\end{frame}

\section{端到端主线：一次成功购物的"黄金路径"}

\begin{frame}{黄金路径：7 跳，每一跳都可能掉链子}
\begin{enumerate}
  \item \textbf{登录} —— 拿到 JWT，后续每个请求带着它过鉴权中间件。
  \item \textbf{看商品} —— 详情走 Cache Aside，热点不打穿 DB。

  \item \textbf{下单} —— 算优惠 $\to$ 预扣库存 $\to$ 一个事务落订单+事件。

\section{贯穿全局的硬骨头：超卖·一致·限流·降级·可观测}

\begin{frame}[fragile]{硬骨头 1：库存防超卖 —— 为什么不用 \texttt{UPDATE WHERE stock>0}}
\begin{itemize}
  \item 高并发下行锁排队 + 尾延迟爆炸；下单链路最怕在事务里等锁。
  \item gomall 把库存搬到 Redis，用 \textbf{两个桶}：\texttt{available}（可卖）/ \texttt{reserved}（已占）。
  \item 预扣 / 提交 / 释放都是 \textbf{单个 Lua 脚本原子完成}，一次 RTT，天然不超卖。
\end{itemize}
\begin{lstlisting}[language=LuaRedis,caption={reserveScript：available 与 reserved 两个 key 原子流转}]
-- KEYS[1]=available 桶  KEYS[2]=reserved 桶  ARGV[1]=n
local avail = redis.call('GET', KEYS[1])
if avail == false then return -2 end           -- 库存未初始化
if tonumber(avail) < tonumber(ARGV[1]) then
    return -1                                  -- 不够，拒单
end
redis.call('DECRBY', KEYS[1], ARGV[1])         -- available -= n
redis.call('INCRBY', KEYS[2], ARGV[1])         -- reserved  += n
return 1
\end{lstlisting}

\end{frame}

\begin{frame}[fragile]{幂等中间件：755K 次重放只成一单}
  \keypoint{业务痛点：用户手抖狂点"提交订单"、客户端超时自动重试、弱网下 App 补发请求——同一笔下单不能变成三笔。}
  \begin{columns}[T]
    \column{0.43\textwidth}
      \textbf{请求态状态机（Redis Hash）}
      \begin{itemize}\small
        \item \texttt{init}：已发券、未执行
        \item \texttt{processing}：首单执行中，后到一律挡回
        \item \texttt{done}：已完成，缓存响应直接回放
      \end{itemize}
      一段 Lua 单次 RTT 原子完成"读态 + 抢锁"。
    \column{0.57\textwidth}
\begin{lstlisting}[language=LuaRedis,caption={acquire：init 到 processing 原子推进}]
local v = redis.call('HGET', KEYS[1], 'state')
if v == false then return {0, ''} end
if v == 'done' then return {2,
    redis.call('HGET', KEYS[1], 'result')} end
if v == 'processing' then return {3, ''} end
-- init: 抢到执行权
redis.call('HSET', KEYS[1], 'state', 'processing')
redis.call('EXPIRE', KEYS[1], tonumber(ARGV[1]))
return {1, ''}
\end{lstlisting}
  \end{columns}
